def all
  identity = Identity.from_omniauth(request.env["omniauth.auth"])
  user = identity.find_or_create_user(current_user_if_exists)

  if user.valid?
    flash.notice = "Signed in!"
    sign_in_and_redirect user
  else
    sign_in user
    redirect_to edit_user_registration_url
  end
end
